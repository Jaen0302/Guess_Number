  <h1>Guess the Number Game</h1>

  <p>I have chosen a number between 1 and 100. You have 5 chances. Good luck!</p>


  <input type="number" id="guessInput" placeholder="Enter your guess">

  <button onclick="checkGuess()">Guess</button>

  <p id="result"></p>

  <p id="chances"></p>


  <script>

    const secretNumber = 47;  // fixed number

    let chancesLeft = 5;


    function checkGuess() {

      const guess = Number(document.getElementById("guessInput").value);

      const result = document.getElementById("result");

      const chances = document.getElementById("chances");


      if (chancesLeft <= 0) {

        result.textContent = "Game over! You already used all your chances.";

        return;

      }


      chancesLeft--;


      if (guess === secretNumber) {

        result.textContent = "🎉 Correct! You guessed the number!";

        chances.textContent = "";

      } else if (chancesLeft === 0) {

        result.textContent = "❌ Wrong! You failed. The number was " + secretNumber + ".";

        chances.textContent = "No chances left.";

      } else if (guess < secretNumber) {

        result.textContent = "Too low! Guess higher.";

        chances.textContent = "Chances left: " + chancesLeft;

      } else {

        result.textContent = "Too high! Guess lower.";

        chances.textContent = "Chances left: " + chancesLeft;

      }

    }

  </script>
